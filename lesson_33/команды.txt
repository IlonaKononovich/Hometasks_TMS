# Запуск Python-скрипта для обработки Excel-файла
python3 process_xls.py input_file.xlsx  
# ^^^^ интерпретатор Python 3
#        ^^^^^^^^^^^^^ имя скрипта
#                        ^^^^^^^^^^^^^^^ входной Excel-файл, передаётся как аргумент

# Сборка Docker-образа с текущими файлами
docker build -t xlsx-processor .  
# ^^^^^^ команда на сборку Docker-образа
#        ^^^^^^^^^^^^^ имя, под которым будет сохранён образ (тег)
#                      ^ текущая директория содержит Dockerfile

# Запуск контейнера из собранного образа в фоновом режиме
docker run -d --name xlsx_processor xlsx-processor  
# ^^^^^^^ запускает контейнер
#         ^ запуск в фоновом (detached) режиме
#              ^^^^^^^^^^^^^ имя контейнера
#                            ^^^^^^^^^^^^^ имя Docker-образа, из которого создаётся контейнер

# Копирование папки /app/data из контейнера в текущую директорию хоста
docker cp xlsx_processor:/app/data .  
# ^^^^^^^ копирует файлы/папки между контейнером и хостом
#         ^^^^^^^^^^^^^ имя контейнера
#                      ^^^^^^^^^^ путь внутри контейнера
#                                ^ путь на хосте, куда будет скопировано (текущая директория)
